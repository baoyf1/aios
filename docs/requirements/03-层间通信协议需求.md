## 一、协议概述

### 1.1 通信规则

```
┌─────────────────────────────────────────────────────────────────────┐
│                         层间通信规则                                  │
│                                                                      │
│  向下通信: 封装 (Encapsulation)                                      │
│    - 添加每层的控制信息                                               │
│    - 从应用层向下传递，层层封装                                       │
│    - 每层只与相邻的下层交互                                          │
│                                                                      │
│  向上通信: 解包 (Decapsulation)                                      │
│    - 剥离控制信息，还原数据                                         │
│    - 从硬件层向上传递，层层解包                                      │
│    - 每层只与相邻的上层交互                                          │
│                                                                      │
│  重要原则:                                                           │
│    - 禁止跨层直接通信                                               │
│    - 只允许邻层交互                                                  │
│    - 每层只添加/剥离自己的控制信息                                   │
│                                                                      │
└─────────────────────────────────────────────────────────────────────┘
```

### 1.2 完整封装/解包流程

```
                    ▲ 向上解包 (Decapsulation)                    
                    │                                              
┌─────────────────────────────────────────────────────────────────────┐
│                        用户输入                                     │

                    ┌─────────────────────────────────────────────┐
                    │ 1. 权限检查 (PathGuard + CapChecker)       │
                    │ 2. 调用对应 Driver (FS_Driver)             │

                    │      request_id: 12345,                    │
                    │      caller_pid: 1,                        │
                    │      syscall_id: SYS_FILE_LIST,            │
                    │      resource_limits: {...}                │
                    │    }                                       │
                    └─────────────────────────────────────────────┘
                                  │ 封装: 添加KernelHeader
                                  ▼
┌─────────────────────────────────────────────────────────────────────┐
                    Layer 2: 内核层
                    ┌─────────────────────────────────────────────┐
                    │ 1. O(1)调度选择下一个Agent                  │
                    │ 2. 分配时间片                               │
                    │ 3. 记录审计日志                             │
                    │ 4. 上下文切换                              │
                    │ 5. 添加 KernelRequest{                     │
                    │      from_pid: 1,                         │
                    │      to_pid: 0,                           │
                    │      priority: HIGH,                       │
                    │      time_slice_ms: 100                   │
                    │    }                                       │
                    └─────────────────────────────────────────────┘
                                  │ 封装: 添加HardwareRequest
                                  ▼
┌─────────────────────────────────────────────────────────────────────┐
                    Layer 1: 硬件层
                    ┌─────────────────────────────────────────────┐
                    │ 1. HTTP调用外部LLM API (如有)              │
                    │ 2. 本地Ollama向量嵌入 (记忆检索)            │

                                  │
                                  │ 结果返回 (逆向逐层解包)
                                  ▼
                    ... 逐层向上返回结果 ...
```

## 二、协议头详细定义

### 2.1 Layer 1: Hardware Request

```cpp
// ==================== hardware_request.h ====================
#pragma once

#include <cstdint>
#include <string>

namespace aios {

// 硬件层操作类型
enum class HardwareOperation : uint32_t {
    HTTP_CALL = 1,        // HTTP请求
    FILE_OPEN = 2,        // 打开文件
    FILE_READ = 3,        // 读取文件
    FILE_WRITE = 4,       // 写入文件
    FILE_CLOSE = 5,       // 关闭文件
    EMBED = 6,            // 向量嵌入
    VECTOR_SEARCH = 7,    // 向量搜索
    MSG_SEND = 8,         // 发送消息
    MSG_RECV = 9,         // 接收消息
};

// 硬件层请求
struct HardwareRequest {
    uint64_t request_id;              // 请求唯一标识 (全局递增)
    HardwareOperation operation;       // 操作类型
    void* raw_params;                  // 原始参数 (类型安全使用需转换)
    void* raw_result;                  // 原始结果
    
    // 构造函数
    HardwareRequest() 
        : request_id(0), operation(HardwareOperation::HTTP_CALL),
          raw_params(nullptr), raw_result(nullptr) {}
    
    HardwareRequest(uint64_t id, HardwareOperation op)
        : request_id(id), operation(op), 
          raw_params(nullptr), raw_result(nullptr) {}
};

// 硬件层响应状态
enum class HardwareStatus : uint32_t {
    SUCCESS = 0,
    ERROR = 1,
    TIMEOUT = 2,
    NOT_FOUND = 3,
    PERMISSION_DENIED = 4,
};

// 硬件层响应
struct HardwareResponse {
    uint64_t request_id;              // 对应请求的ID
    HardwareStatus status;            // 状态
    int error_code;                   // 错误码 (0表示成功)
    std::string error_message;         // 错误信息
    void* result;                     // 结果指针
    
    static HardwareResponse success(uint64_t req_id, void* result = nullptr) {
        return {req_id, HardwareStatus::SUCCESS, 0, "", result};
    }
    
    static HardwareResponse error(uint64_t req_id, int code, 
                                   const std::string& msg) {
        return {req_id, HardwareStatus::ERROR, code, msg, nullptr};
    }
};

} // namespace aios
```

### 2.2 Layer 2: Kernel Request

```cpp
// ==================== kernel_request.h ====================
#pragma once

#include "hardware_request.h"
#include <cstdint>
#include <string>

namespace aios {

// 进程ID类型
using pid_t = uint32_t;
constexpr pid_t PID_INVALID = 0;
constexpr pid_t PID_CENTRAL = 0;  // 中央Agent PID

// Agent优先级
enum class AgentPriority : uint32_t {
    IDLE = 0,        // 空闲 (不会被调度)
    LOW = 1,         // 低优先级
    NORMAL = 2,     // 普通优先级
    HIGH = 3,       // 高优先级
    CRITICAL = 4,   // 关键优先级
    SYSTEM = 5,     // 系统优先级 (最高)
};

// 调度算法类型
enum class SchedulerType : uint32_t {
    O1 = 1,          // O(1)调度器
    FIFO = 2,       // 先进先出
    ROUND_ROBIN = 3, // 时间片轮转
};

// 内核请求
struct KernelRequest {
    uint64_t request_id;              // 请求唯一标识
    pid_t from_pid;                   // 来源Agent PID
    pid_t to_pid;                     // 目标Agent PID (0=中央Agent)
    AgentPriority priority;           // 调度优先级
    uint32_t time_slice_ms;           // 分配的时间片 (毫秒)
    HardwareRequest hardware;         // 下层硬件请求
    uint64_t deadline_ms;             // 截止时间 (0表示无截止)
    uint64_t created_at;              // 创建时间戳
    
    KernelRequest()
        : request_id(0), from_pid(PID_INVALID), to_pid(PID_INVALID),
          priority(AgentPriority::NORMAL), time_slice_ms(100),
          deadline_ms(0), created_at(0) {}
    
    KernelRequest(uint64_t id, pid_t from, pid_t to)
        : request_id(id), from_pid(from), to_pid(to),
          priority(AgentPriority::NORMAL), time_slice_ms(100),
          deadline_ms(0), created_at(get_current_timestamp()) {}
};

// 进程控制块
struct ProcessControlBlock {
    pid_t pid;                        // 进程ID
    std::string name;                 // 进程名称
    AgentPriority priority;           // 当前优先级
    AgentPriority static_priority;    // 静态优先级
    uint64_t time_used_ms;            // 已用时间
    uint32_t time_slice_ms;           // 分配的时间片
    uint64_t last_scheduled_at;      // 上次调度时间
    enum class State { NEW, READY, RUNNING, WAITING, TERMINATED } state;
    pid_t parent_pid;                 // 父进程PID
    
    ProcessControlBlock(pid_t p, const std::string& n)
        : pid(p), name(n), priority(AgentPriority::NORMAL),
          static_priority(AgentPriority::NORMAL), time_used_ms(0),
          time_slice_ms(100), last_scheduled_at(0), 
          state(State::NEW), parent_pid(PID_INVALID) {}
};

} // namespace aios
```

```cpp
struct ResourceLimits {
    uint64_t max_memory_bytes;        // 最大内存 (字节)
    uint64_t max_cpu_time_ms;         // 最大CPU时间 (毫秒)
    uint64_t max_file_size_bytes;     // 最大文件大小
    uint32_t max_open_files;          // 最大打开文件数
    uint32_t max_network_bandwidth_bytes; // 最大网络带宽
    
    ResourceLimits()
        : max_memory_bytes(1024 * 1024 * 1024),  // 1GB
          max_cpu_time_ms(60000),                // 1分钟
          max_file_size_bytes(100 * 1024 * 1024), // 100MB
          max_open_files(100),
          max_network_bandwidth_bytes(0) {}       // 无限制

// 权限上下文
struct PermissionContext {
    pid_t agent_id;                   // Agent ID
    uint64_t capabilities;             // Capability位掩码
    std::vector<std::string> allowed_paths;   // 允许的路径
    std::vector<std::string> denied_paths;    // 拒绝的路径
    ResourceLimits limits;            // 资源限制
    
    bool has_capability(uint32_t cap) const {
        return (capabilities & (1ULL << cap)) != 0;
    }
};

// Variant类型用于参数
using SysCallArg = std::variant<
    int64_t, 
    double, 
    std::string, 
    std::vector<uint8_t>,
    std::vector<std::string>
>;

// 系统调用请求
struct SysCallRequest {
    uint64_t trace_id;               // 追踪ID (贯穿整个调用链)
    uint64_t request_id;             // 请求唯一标识
    pid_t caller_pid;                // 调用者Agent PID
    uint32_t syscall_id;             // 系统调用号
    std::vector<SysCallArg> args;    // 参数列表
    ResourceLimits limits;          // 资源限制
    PermissionContext perms;        // 权限上下文
    KernelRequest kernel;            // 内核请求
    
    SysCallRequest()
        : trace_id(0), request_id(0), caller_pid(PID_INVALID),
          syscall_id(0), limits(), perms() {}
};

// 系统调用结果状态
enum class SysCallStatus : uint32_t {
    OK = 0,
    ERROR = 1,
    PERMISSION_DENIED = 2,
    NOT_FOUND = 3,
    TIMEOUT = 4,
    RESOURCE_EXCEEDED = 5,
    INVALID_ARGUMENT = 6,
};

// 系统调用结果
struct SysCallResult {
    enum class Status { OK, ERROR };
    Status status;
    int error_code;
    std::string error_msg;
    SysCallArg return_value;
    
    static SysCallResult ok(SysCallArg&& value) {
        return {Status::OK, 0, "", std::move(value)};
    }
    
    static SysCallResult ok() {
        return {Status::OK, 0, "", {}};
    }
    
    static SysCallResult error(int code, const std::string& msg) {
        return {Status::ERROR, code, msg, {}};
    }
};

} // namespace aios
```

### 2.4 Layer 4: Middleware

```cpp
// ==================== middleware_message.h ====================
#pragma once

#include "syscall_request.h"
#include <string>
#include <vector>

namespace aios {

// 消息类型
enum class MessageType : uint32_t {
    REQUEST = 1,         // 请求
    RESPONSE = 2,        // 响应
    NOTIFICATION = 3,    // 通知
    HEARTBEAT = 4,       // 心跳
    ERROR = 5,           // 错误
};

// 中间件消息
struct MiddlewareMessage {
    uint64_t trace_id;           // 追踪ID (贯穿整个调用链)
    pid_t agent_id;              // 来源Agent
    uint64_t timestamp;         // 时间戳 (毫秒)
    std::string session_id;     // 会话ID
    MessageType type;            // 消息类型
    std::string payload;         // JSON序列化数据
    SysCallRequest syscall;      // 系统调用请求
    uint32_t sequence;           // 序列号 (用于排序)
    uint32_t flags;              // 标志位
    
    MiddlewareMessage()
        : trace_id(0), agent_id(PID_INVALID), timestamp(0),
          type(MessageType::REQUEST), sequence(0), flags(0) {}
    
    // 创建请求消息
    static MiddlewareMessage create_request(
        pid_t from,
        const std::string& session,
        uint64_t trace,
        const SysCallRequest& sys_req
    ) {
        MiddlewareMessage msg;
        msg.agent_id = from;
        msg.session_id = session;
        msg.trace_id = trace;
        msg.type = MessageType::REQUEST;
        msg.syscall = sys_req;
        msg.timestamp = get_current_timestamp_ms();
        return msg;
    }
};

// 会话状态
enum class SessionStatus : uint32_t {
    ACTIVE = 1,          // 活跃
    IDLE = 2,            // 空闲
    EXPIRING = 3,        // 即将过期
    EXPIRED = 4,         // 已过期
    CLOSED = 5,          // 已关闭
};

// 会话信息
struct Session {
    std::string session_id;      // 会话ID
    pid_t agent_id;              // 关联的Agent
    SessionStatus status;        // 状态
    uint64_t created_at;        // 创建时间
    uint64_t last_active_at;    // 最后活跃时间
    uint64_t expires_at;        // 过期时间
    uint32_t message_count;     // 消息计数
    std::vector<std::string> context;  // 上下文摘要
    
    Session(const std::string& id, pid_t agent, uint64_t ttl_ms = 3600000)
        : session_id(id), agent_id(agent), status(SessionStatus::ACTIVE),
          created_at(get_current_timestamp_ms()),
          last_active_at(created_at),
          expires_at(created_at + ttl_ms),
          message_count(0) {}
    
    bool is_expired() const {
        return get_current_timestamp_ms() > expires_at;
    }
};

} // namespace aios
```

### 2.5 Layer 5: Application

```cpp
// ==================== app_request.h ====================
#pragma once

#include "middleware_message.h"
#include <string>
#include <vector>
#include <variant>

namespace aios {

// 实体类型
enum class EntityType : uint32_t {
    TEXT = 1,           // 文本
    PATH = 2,          // 路径
    FILE = 3,          // 文件
    URL = 4,           // URL
    EMAIL = 5,         // 邮箱
    USER = 6,          // 用户
    TIME = 7,          // 时间
    NUMBER = 8,        // 数字
    KEYWORD = 9,       // 关键词
    COMMAND = 10,       // 命令
};

// 实体
struct Entity {
    EntityType type;            // 实体类型
    std::string value;           // 实体值
    float confidence;            // 置信度
    std::string raw_text;        // 原始文本
    
    Entity(EntityType t, const std::string& v, float c = 1.0f)
        : type(t), value(v), confidence(c), raw_text(v) {}
};

// 意图
struct Intent {
    std::string raw_text;        // 原始文本
    std::string domain;          // 领域: "file", "system", "mail", "code", etc.
    std::string action;          // 动作: "organize", "send", "search", etc.
    std::vector<Entity> entities; // 实体列表
    float confidence;            // 置信度
    std::string user_id;         // 用户ID
    std::string trace_id;        // 追踪ID
    
    Intent() : confidence(0.0), trace_id("") {}
    
    bool is_confident() const { return confidence >= 0.7f; }
    
    std::string to_string() const {
        return std::string("Intent[") + domain + ":" + action + 
               ", confidence=" + std::to_string(confidence) + "]";
    }
};

// 任务优先级
enum class TaskPriority : uint32_t {
    LOW = 1,
    NORMAL = 2,
    HIGH = 3,
    URGENT = 4,
};

// 子任务
struct SubTask {
    std::string task_id;                     // 子任务ID
    std::string agent_domain;               // 目标Agent领域
    std::string description;                // 任务描述
    std::variant<std::string, std::map<std::string, std::string>> input;  // 输入数据
    std::vector<std::string> dependencies;  // 依赖的子任务ID
    TaskPriority priority;                  // 优先级
    uint64_t timeout_ms;                   // 超时时间
    std::string trace_id;                  // 追踪ID
    
    SubTask() 
        : priority(TaskPriority::NORMAL), 
          timeout_ms(60000),  // 默认1分钟
          trace_id("") {}
    
    SubTask(const std::string& id, const std::string& domain)
        : task_id(id), agent_domain(domain),
          priority(TaskPriority::NORMAL),
          timeout_ms(60000), trace_id("") {}
};

// 应用层请求
struct AppRequest {
    std::string raw_input;           // 原始用户输入
    Intent parsed_intent;          // 解析后的意图
    std::vector<SubTask> tasks;     // 子任务列表
    MiddlewareMessage middleware;   // 中间件消息
    std::string user_id;            // 用户ID
    std::string session_id;        // 会话ID
    std::string locale;             // 语言区域
    uint64_t timestamp;             // 时间戳
    
    AppRequest() : timestamp(0) {}
    
    static AppRequest create(const std::string& input, const Intent& intent) {
        AppRequest req;
        req.raw_input = input;
        req.parsed_intent = intent;
        req.timestamp = get_current_timestamp_ms();
        return req;
    }
};

// 任务结果状态
enum class ResultStatus : uint32_t {
    PENDING = 1,       // 待处理
    RUNNING = 2,       // 执行中
    SUCCESS = 3,       // 成功
    PARTIAL = 4,       // 部分成功
    FAILED = 5,        // 失败
    CANCELLED = 6,     // 取消
    TIMEOUT = 7,       // 超时
};

// 任务结果
struct TaskResult {
    bool success;               // 是否成功
    std::string output;         // 执行结果
    std::string error;          // 错误信息
    std::string trace_id;       // 追踪ID
    uint64_t duration_ms;       // 执行耗时
    ResultStatus status;        // 状态
    std::string task_id;        // 任务ID
    std::map<std::string, std::string> metadata;  // 元数据
    
    TaskResult() 
        : success(false), duration_ms(0), 
          status(ResultStatus::PENDING) {}
    
    static TaskResult success_result(const std::string& tid, 
                                     const std::string& out,
                                     uint64_t duration,
                                     const std::string& trace) {
        return {true, out, "", trace, duration, 
                ResultStatus::SUCCESS, tid, {}};
    }
    
    static TaskResult error_result(const std::string& tid,
                                    const std::string& err,
                                    const std::string& trace) {
        return {false, "", err, trace, 0, 
                ResultStatus::FAILED, tid, {}};
    }
};
```

### 3.2 Intent 解析流程

```python
class IntentParser:
    """意图解析器"""
    
    def __init__(self, llm):
        self.llm = llm
        self.patterns = [
            IntentPattern("整理", "file", "organize"),
            IntentPattern("移动", "file", "move"),
            IntentPattern("复制", "file", "copy"),
            IntentPattern("删除", "file", "delete"),
            IntentPattern("发送", "mail", "send"),
            IntentPattern("搜索", "search", "search"),
            # ... 更多模式
        ]
    
    async def parse(self, user_input: str, trace_id: str) -> Intent:
        # 1. 快速模式匹配
        for pattern in self.patterns:
            if pattern.matches(user_input):
                intent = Intent(
                    raw_text=user_input,
                    domain=pattern.domain,
                    action=pattern.action,
                    confidence=0.9,
                    trace_id=trace_id
                )
                intent.entities = self._extract_entities(user_input, pattern)
                return intent
        
        # 2. LLM深度解析
        prompt = f"""
分析用户输入，提取意图信息：

用户输入: {user_input}

请返回JSON格式的意图分析结果，包含：
- domain: 领域 (file/mail/code/search/note/data/media/system/network)
- action: 动作 (list/read/write/send/search/organize/create/delete/...)
- entities: 实体列表 [{type, value, confidence}]
- user_id: 用户标识 (从上下文获取)
- confidence: 置信度 (0-1)

只返回JSON，不要有其他内容。
"""
        response = await self.llm.agenerate([prompt])
        return Intent.from_json(response.text, trace_id)
    
    def _extract_entities(self, text: str, pattern: IntentPattern) -> List[Entity]:
        """提取实体"""
        entities = []
        
        # 路径提取
        path_pattern = r'[/]?[\w]+(?:/[\w]+)*'
        for match in re.finditer(path_pattern, text):
            entities.append(Entity(EntityType.PATH, match.group(), 0.95))
        
        # 关键词提取
        keywords = ["PDF", "文档", "文件", "图片", "视频"]
        for kw in keywords:
            if kw in text:
                entities.append(Entity(EntityType.KEYWORD, kw, 0.9))
        
        return entities
```

```python
class TaskPlanner:
    """任务规划器"""
    
    def __init__(self, llm):
        self.llm = llm
    
    async def plan(self, intent: Intent, trace_id: str) -> List[SubTask]:
        # 简单意图直接返回
        if self._is_simple_intent(intent):
            return [self._simple_task(intent, trace_id)]
        
        # 复杂意图需要LLM规划
        prompt = f"""
分析以下意图，分解为可执行的子任务：

意图: {intent.to_string()}
输入: {intent.raw_text}

任务类型定义:
- file: list/read/write/move/copy/delete/search/organize/stat/mkdir/rmdir
- mail: send/read/list/search/draft/attachment
- code: read/write/search/test/explain/refactor/deploy
- search: web/knowledge/file/desktop/internal
- note: create/read/update/delete/organize/link/search
- data: query/transform/analyze/report/visualize
- media: process/thumb/transcribe/convert

请返回任务列表JSON数组，每个任务包含:
- task_id: 任务ID (数字)
- domain: 领域
- action: 动作
- input: 输入参数 (dict)
- dependencies: 依赖的任务ID列表
- priority: HIGH/NORMAL/LOW
- timeout_ms: 超时时间 (毫秒)

示例输出:
[
    {{"task_id": "1", "domain": "file", "action": "list", "input": {{"path": "/Desktop"}}, "dependencies": [], "priority": "HIGH", "timeout_ms": 30000}}
]
"""
        response = await self.llm.agenerate([prompt])
        tasks = self._parse_tasks(response.text)
        
        # 补充trace_id
        for task in tasks:
            task.trace_id = trace_id
        
        return tasks
    
    def _is_simple_intent(self, intent: Intent) -> bool:
        """判断是否为简单意图"""
        simple_actions = {"list", "read", "send", "search", "get", "show", "get"}
        return (len(intent.entities) <= 2 and 
                intent.action.lower() in simple_actions)
    
    def _simple_task(self, intent: Intent, trace_id: str) -> SubTask:
        """创建简单任务"""
        task = SubTask("1", intent.domain)
        task.action = intent.action
        task.description = f"{intent.action} {intent.domain}"
        task.input = {"entities": [e.value for e in intent.entities]}
        task.trace_id = trace_id
        task.priority = TaskPriority.NORMAL
        return task
```

```python
class JSONSerializer:
    """JSON序列化器"""
    
    @staticmethod
    def serialize(msg: MiddlewareMessage) -> str:
        """序列化消息为JSON"""
        return json.dumps({
            "trace_id": msg.trace_id,
            "agent_id": msg.agent_id,
            "timestamp": msg.timestamp,
            "session_id": msg.session_id,
            "type": msg.type.name,
            "payload": msg.payload,
            "syscall": {
                "trace_id": msg.syscall.trace_id,
                "request_id": msg.syscall.request_id,
                "caller_pid": msg.syscall.caller_pid,
                "syscall_id": msg.syscall.syscall_id,
                "args": [str(arg) for arg in msg.syscall.args]
            },
            "sequence": msg.sequence,
            "flags": msg.flags
        }, ensure_ascii=False)
    
    @staticmethod
    def deserialize(data: str) -> MiddlewareMessage:
        """从JSON反序列化"""
        obj = json.loads(data)
        msg = MiddlewareMessage()
        msg.trace_id = obj["trace_id"]
        msg.agent_id = obj["agent_id"]
        msg.timestamp = obj["timestamp"]
        msg.session_id = obj["session_id"]
        msg.type = MessageType[obj["type"]]
        msg.payload = obj["payload"]
        msg.syscall = SysCallRequest()  # 简化
        msg.sequence = obj.get("sequence", 0)
        msg.flags = obj.get("flags", 0)
        return msg
```

```python
class TraceManager:
    """trace_id管理器"""
    
    def __init__(self):
        self.traces = {}  # trace_id -> TraceRecord
    
    def create_trace_id(self) -> str:
        """创建新的trace_id"""
        timestamp = int(time.time() * 1000)
        random_hex = secrets.hexlify(4).decode()
        return f"trace_{timestamp}_{random_hex}"
    
    def start_trace(self, trace_id: str, root_task: str):
        """开始追踪"""
        self.traces[trace_id] = TraceRecord(
            trace_id=trace_id,
            root_task=root_task,
            start_time=get_current_timestamp_ms(),
            entries=[]
        )
    
    def add_entry(self, trace_id: str, entry: TraceEntry):
        """添加追踪条目"""
        if trace_id in self.traces:
            self.traces[trace_id].entries.append(entry)
    
    def get_trace(self, trace_id: str) -> TraceRecord:
        """获取追踪记录"""
        return self.traces.get(trace_id)
    
    def visualize_trace(self, trace_id: str) -> str:
        """可视化追踪结果"""
        trace = self.traces.get(trace_id)
        if not trace:
            return f"Trace not found: {trace_id}"
        
        output = f"Trace: {trace_id}\n"
        output += f"Root Task: {trace.root_task}\n"
        output += f"Duration: {trace.duration_ms}ms\n"
        output += "-" * 60 + "\n"
        
        for entry in trace.entries:
            output += f"[{entry.agent_name}] {entry.action}: {entry.details}\n"
        
        return output
```
