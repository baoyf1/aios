## 一、设计原则

```
┌─────────────────────────────────────────────────────────────────────┐
│                         上下文分层设计                               │
├─────────────────────────────────────────────────────────────────────┤
│                                                                      │
│  上层 (Context Layer)                                               │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  业务上下文 ← 组装/销毁/传递 ← 需要额外处理                  │   │
│  │                                                              │   │
│  │  例如:                                                       │   │
│  │  - FileAgentContext: 文件操作相关状态                        │   │
│  │  - MailAgentContext: 邮件相关状态                            │   │
│  │  - SessionContext: 会话相关状态                              │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                              │                                        │
│                              ▼ on_attach() / on_detach()            │
│  上下文桥接层 (Bridge)                                              │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  负责将业务上下文绑定到IO事件                                │   │
│  │                                                              │   │
│  │  - cleanup回调注册                                          │   │
│  │  - 生命周期事件通知                                          │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                              │                                        │
│                              ▼                                        │
│  底层 (Event Engine)                                                │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  epoll ← 异步IO ← fd/事件 ← 不关心上层                     │   │
│  │                                                              │   │
│  │  - 只管理fd和事件                                           │   │
│  │  - 不感知业务逻辑                                           │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                      │
└─────────────────────────────────────────────────────────────────────┘

核心原则：
1. 底层不关心上层上下文 (分离关注)
2. 上层通过cleanup回调处理额外逻辑 (额外处理入口)
3. 底层先调用cleanup再销毁资源 (生命周期安全)
4. 上层负责user_context的分配和释放 (内存管理)
```

## 二、上下文结构设计

### 2.1 底层IO上下文 (IoContext)

```cpp
// ==================== io_context.h ====================
#pragma once

#include <cstdint>
#include <functional>

namespace aios {

// IO事件类型
enum class IoEventType : uint32_t {
    READ = 1,       // 可读
    WRITE = 2,      // 可写
    ERROR = 4,      // 错误
    HANGUP = 8,     // 挂起
};

// IO上下文 (底层，不感知业务)
struct IoContext {
    // 用户上下文指针 (上层业务上下文)
    // 底层不解释这个指针，只是透传
    void* user_context;
    
    // 清理回调 (detach时调用，用于上层执行额外处理)
    // 签名: void cleanup()
    std::function<void()> cleanup;
    
    // 文件描述符
    int fd;
    
    // 监听的事件类型 (EPOLLIN | EPOLLOUT | EPOLLET)
    uint32_t events;
    
    // 创建时间
    uint64_t created_at;
    
    // 最后活跃时间
    uint64_t last_active_at;
    
    IoContext()
        : user_context(nullptr), fd(-1), events(0),
          created_at(0), last_active_at(0) {}
    
    IoContext(void* ctx, int fileDescriptor, uint32_t evts)
        : user_context(ctx), fd(fileDescriptor), events(evts),
          created_at(get_current_timestamp_ms()),
          last_active_at(created_at) {}
    
    // 更新活跃时间
    void touch() {
        last_active_at = get_current_timestamp_ms();
    }
    
    // 检查是否超时
    bool is_idle(uint64_t idle_timeout_ms) const {
        return (get_current_timestamp_ms() - last_active_at) > idle_timeout_ms;
    }
};

} // namespace aios
```

### 2.2 上层业务上下文 (BusinessContext)

```cpp
// ==================== business_context.h ====================
#pragma once

#include <string>
#include <unordered_map>
#include <memory>
#include "io_context.h"

namespace aios {

// 业务上下文基类
// 所有上层业务上下文必须继承此类
class BusinessContext {
public:
    BusinessContext() : created_at_(get_current_timestamp_ms()) {}
    virtual ~BusinessContext() = default;
    
    // 额外处理：装载时 (Attach)
    // 当底层attach时会调用此方法
    // 子类可重写以执行额外的初始化逻辑
    virtual void on_attach() {
        // 默认实现为空
        // 子类可重写，例如：
        // - 加载记忆数据
        // - 初始化状态
        // - 分配资源
    }
    
    // 额外处理：卸载时 (Detach)
    // 当底层detach时会调用此方法
    // 子类可重写以执行额外的清理逻辑
    virtual void on_detach() {
        // 默认实现为空
        // 子类可重写，例如：
        // - 保存状态
        // - 释放资源
        // - 持久化数据
    }
    
    uint64_t created_at() const { return created_at_; }
    
private:
    uint64_t created_at_;
};

// 会话上下文
struct SessionContext : BusinessContext {
    SessionContext(const std::string& session_id, pid_t agent_id)
        : session_id_(session_id), agent_id_(agent_id) {}
    
    std::string session_id_;
    pid_t agent_id_;
    std::unordered_map<std::string, std::string> attributes_;
    
    void on_attach() override {
        // 加载会话数据
    }
    
    void on_detach() override {
        // 保存会话数据
    }
};

// 文件操作上下文
struct FileOperationContext : BusinessContext {
    FileOperationContext(pid_t agent_id, const std::string& path)
        : agent_id_(agent_id), path_(path) {}
    
    pid_t agent_id_;
    std::string path_;
    std::string current_file_;
    size_t file_offset_;
    bool is_modified_;
    
    void on_attach() override {
        // 打开文件，检查权限
    }
    
    void on_detach() override {
        // 关闭文件句柄，保存修改
    }
};

} // namespace aios
```

## 三、EventEngine 实现

### 3.1 EventEngine 核心

```cpp
// ==================== event_engine.h ====================
#pragma once

#include "io_context.h"
#include <unordered_map>
#include <vector>
#include <functional>
#include <thread>
#include <atomic>

namespace aios {

// IO事件回调
using IoEventCallback = std::function<void(int fd, uint32_t events)>;

// EventEngine - 底层事件引擎
// 负责管理fd和事件，不感知上层业务
class EventEngine {
public:
    EventEngine();
    ~EventEngine();
    
    // 禁止拷贝
    EventEngine(const EventEngine&) = delete;
    EventEngine& operator=(const EventEngine&) = delete;
    
    // 启动事件循环
    void start();
    
    // 停止事件循环
    void stop();
    
    // attach: 装载上下文到底层
    // 流程:
    // 1. 创建IoContext
    // 2. 注册fd到epoll
    // 3. 调用上层on_attach()
    // 返回: IoContext句柄
    IoContext* attach(int fd, void* user_ctx, std::function<void()> cleanup,
                      uint32_t events = EPOLLIN | EPOLLOUT | EPOLLET);
    
    // detach: 卸载上下文
    // 流程:
    // 1. 调用cleanup() - 通知上层执行额外处理
    // 2. 从epoll移除fd
    // 3. 销毁IoContext
    // 4. 上层on_detach()已在cleanup中执行
    int detach(int fd);
    
    // 获取用户上下文
    void* get_user_context(int fd);
    
    // 注册IO事件回调
    void set_event_callback(IoEventCallback cb);
    
    // 修改监听事件
    int modify(int fd, uint32_t events);
    
private:
    void run_loop();
    void process_events(int timeout_ms);
    
    int epfd_;                                    // epoll fd
    std::unordered_map<int, IoContext*> ctx_map_; // fd -> context
    std::atomic<bool> running_;                  // 运行标志
    std::thread loop_thread_;                     // 事件循环线程
    IoEventCallback event_callback_;              // 事件回调
};

} // namespace aios
```

### 3.2 EventEngine 实现细节

```cpp
// ==================== event_engine.cpp ====================
#include "event_engine.h"
#include <sys/epoll.h>
#include <unistd.h>
#include <cstring>
#include <fcntl.h>

namespace aios {

EventEngine::EventEngine() : epfd_(-1), running_(false) {
    // 创建epoll fd
    epfd_ = epoll_create1(EPOLL_CLOEXEC);
    if (epfd_ < 0) {
        throw std::runtime_error("Failed to create epoll fd");
    }
}

EventEngine::~EventEngine() {
    stop();
    if (epfd_ >= 0) {
        close(epfd_);
    }
}

void EventEngine::start() {
    if (running_) return;
    running_ = true;
    loop_thread_ = std::thread(&EventEngine::run_loop, this);
}

void EventEngine::stop() {
    if (!running_) return;
    running_ = false;
    if (loop_thread_.joinable()) {
        loop_thread_.join();
    }
}

IoContext* EventEngine::attach(int fd, void* user_ctx, 
                                std::function<void()> cleanup,
                                uint32_t events) {
    // 1. 创建IoContext
    auto* ctx = new IoContext(user_ctx, fd, events);
    
    // 2. 设置文件描述符为非阻塞
    int flags = fcntl(fd, F_GETFL, 0);
    fcntl(fd, F_SETFL, flags | O_NONBLOCK);
    
    // 3. 注册到epoll
    epoll_event ev;
    ev.events = events;
    ev.data.ptr = ctx;
    
    if (epoll_ctl(epfd_, EPOLL_CTL_ADD, fd, &ev) < 0) {
        delete ctx;
        return nullptr;
    }
    
    // 4. 保存到map
    ctx_map_[fd] = ctx;
    
    // 5. 调用上层on_attach()
    if (user_ctx) {
        auto* biz_ctx = static_cast<BusinessContext*>(user_ctx);
        biz_ctx->on_attach();
    }
    
    return ctx;
}

int EventEngine::detach(int fd) {
    auto it = ctx_map_.find(fd);
    if (it == ctx_map_.end()) {
        return -1;  // 未找到
    }
    
    IoContext* ctx = it->second;
    
    // 1. 调用cleanup() - 通知上层执行额外处理
    // 这是关键：cleanup中会调用上层on_detach()
    if (ctx->cleanup) {
        ctx->cleanup();
    }
    
    // 2. 从epoll移除
    epoll_ctl(epfd_, EPOLL_CTL_DEL, fd, nullptr);
    
    // 3. 从map移除
    ctx_map_.erase(it);
    
    // 4. 销毁IoContext
    delete ctx;
    
    return 0;
}

void* EventEngine::get_user_context(int fd) {
    auto it = ctx_map_.find(fd);
    if (it == ctx_map_.end()) {
        return nullptr;
    }
    return it->second->user_context;
}

void EventEngine::set_event_callback(IoEventCallback cb) {
    event_callback_ = cb;
}

int EventEngine::modify(int fd, uint32_t events) {
    auto it = ctx_map_.find(fd);
    if (it == ctx_map_.end()) {
        return -1;
    }
    
    it->second->events = events;
    
    epoll_event ev;
    ev.events = events;
    ev.data.ptr = it->second;
    
    return epoll_ctl(epfd_, EPOLL_CTL_MOD, fd, &ev);
}

void EventEngine::run_loop() {
    while (running_) {
        process_events(100);  // 100ms超时
    }
}

void EventEngine::process_events(int timeout_ms) {
    constexpr int MAX_EVENTS = 64;
    epoll_event events[MAX_EVENTS];
    
    int n = epoll_wait(epfd_, events, MAX_EVENTS, timeout_ms);
    if (n < 0) {
        if (errno == EINTR) return;
        return;
    }
    
    for (int i = 0; i < n; ++i) {
        auto* ctx = static_cast<IoContext*>(events[i].data.ptr);
        
        // 更新活跃时间
        ctx->touch();
        
        // 调用事件回调
        if (event_callback_) {
            event_callback_(ctx->fd, events[i].events);
        }
        
        // 检查挂起事件
        if (events[i].events & EPOLLHUP) {
            // 触发detach
            detach(ctx->fd);
        }
    }
}

} // namespace aios
```

## 四、装载卸载流程

### 4.1 Attach 流程

```
┌─────────────────────────────────────────────────────────────────────┐
│                         Attach 完整流程                              │
├─────────────────────────────────────────────────────────────────────┤
│                                                                      │
│  1. 上层创建 BusinessContext                                        │
│     └─> auto* biz_ctx = new SessionContext(session_id, agent_id)   │
│                                                                      │
│  2. 上层准备cleanup回调                                             │
│     └─> auto cleanup = [biz_ctx]() {                               │
│             biz_ctx->on_detach();  // 执行额外清理                   │
│             delete biz_ctx;        // 销毁业务上下文                │
│         };                                                          │
│                                                                      │
│  3. 调用底层attach()                                                │
│     └─> engine.attach(fd, biz_ctx, cleanup, events);              │
│                                                                      │
│  4. 底层创建IoContext，绑定business_ctx                            │
│     └─> IoContext{user_context=biz_ctx, cleanup=cleanup, ...}    │
│                                                                      │
│  5. 底层注册fd到epoll                                               │
│     └─> epoll_ctl(EPOLL_CTL_ADD)                                   │
│                                                                      │
│  6. 底层调用上层on_attach()                                         │
│     └─> biz_ctx->on_attach()                                       │
│     └─> 业务特定初始化: 加载记忆、初始化状态、分配资源              │
│                                                                      │
│  7. 返回IoContext句柄给上层                                         │
│     └─> return io_ctx;                                             │
│                                                                      │
│  ✅ attach完成                                                       │
│                                                                      │
└─────────────────────────────────────────────────────────────────────┘
```

```cpp
        // 先执行自身的初始化
        // ...
    }
    
    void on_detach() override {
        // 1. 先执行自身的清理
```

## 六、设计要点总结

| 要点 | 说明 |
|------|------|
| **分离关注** | 底层只管IO事件，上层管业务逻辑 |
| **额外处理入口** | cleanup回调 + on_attach/on_detach |
| **生命周期安全** | detach时cleanup在底层删除前执行 |
| **内存管理** | 上层负责user_context的分配和释放 |
| **多线程安全** | epoll在单一线程，事件分发到业务线程池 |
| **嵌套支持** | NestedContext支持跨Agent上下文借用 |
| **可扩展性** | 新增业务上下文只需继承BusinessContext |

### 7.1 完整使用示例

```cpp
// ==================== usage_example.cpp ====================
#include "event_engine.h"
#include "business_context.h"
#include <iostream>
#include <sys/socket.h>

// 自定义业务上下文
class MyConnectionContext : public BusinessContext {
public:
    MyConnectionContext(pid_t agent_id, int client_fd)
        : agent_id_(agent_id), client_fd_(client_fd) {
        std::cout << "MyConnectionContext created" << std::endl;
    }
    
    ~MyConnectionContext() {
        std::cout << "MyConnectionContext destroyed" << std::endl;
    }
    
    void on_attach() override {
        std::cout << "Connection attached, agent_id=" << agent_id_ << std::endl;
        // 初始化连接状态
        // 加载相关记忆
    }
    
    void on_detach() override {
        std::cout << "Connection detached, agent_id=" << agent_id_ << std::endl;
        // 保存状态
        // 清理资源
    }
    
    pid_t agent_id() const { return agent_id_; }
    int client_fd() const { return client_fd_; }
    
private:
    pid_t agent_id_;
    int client_fd_;
};

class MyServer {
public:
    MyServer() : engine_() {
        engine_.set_event_callback([this](int fd, uint32_t events) {
            handle_event(fd, events);
        });
    }
    
    void start(int listen_fd) {
        engine_.start();
        
        // attach监听socket
        auto* ctx = new SessionContext("listener", 0);
        engine_.attach(listen_fd, ctx, [ctx]() {
            ctx->on_detach();
            delete ctx;
        });
        
        // 保持运行
        std::this_thread::sleep_for(std::chrono::hours(24));
    }
    
private:
    void handle_event(int fd, uint32_t events) {
        if (events & EPOLLIN) {
            // 处理新连接
            int client_fd = accept(fd, nullptr, nullptr);
            if (client_fd >= 0) {
                auto* conn_ctx = new MyConnectionContext(1, client_fd);
                engine_.attach(client_fd, conn_ctx, [conn_ctx]() {
                    conn_ctx->on_detach();
                    delete conn_ctx;
                });
            }
        }
    }
    
    EventEngine engine_;
};
```
