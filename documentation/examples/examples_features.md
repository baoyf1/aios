### 5.2 Agent开发示例
```python
# 自定义Agent示例
from aios.agents import BaseAgent

class CustomAgent(BaseAgent):
    """自定义Agent示例"""
    
    def __init__(self, config):
        super().__init__(config)
        self.name = "CustomAgent"
    
    def process_task(self, task):
        """处理任务"""
        # 实现任务处理逻辑
        return f"CustomAgent处理任务: {task}"
    
    def get_capabilities(self):
        """获取Agent能力"""
        return ["custom_task"]

# 注册Agent
from aios.agents import AgentRegistry
AgentRegistry.register(CustomAgent)
```

### 5.3 工具使用示例
```python
# 文件操作示例
from aios.tools import FileOps

# 初始化文件操作工具
file_ops = FileOps()

# 读取文件
content = file_ops.read_file("C:\\example.txt")
print(content)

# 写入文件
file_ops.write_file("C:\\example.txt", "Hello, AIOS!")

# 搜索文件
files = file_ops.search_files("C:\\", "*.txt")
print(files)
```
